```java
    public ListNode reverseList(ListNode head) {
        if (head == null || head.next == null) return head;
        ListNode newHead = reverseList(head.next);
        head.next.next = head;
        head.next = null;
        return newHead;
    }
}
```

```js
var reverseList = function(head) {
    if (head == null) return null
    let cur = head.next
    head.next = null
    while (cur != null) {
        const next = cur.next
        cur.next = head
        head = cur
        cur = next
    }
    return head
};
```

**复杂度分析**

- 时间复杂度：$O(n)$。
- 空间复杂度：迭代做法为 $O(1)$，递归做法为 $O(n)$。
